# Read CSV with schema
from pyspark.sql.types import StructType, StructField, StringType, IntegerType
schema = StructType([StructField("id", IntegerType()), StructField("name", StringType())])
df = spark.read.csv("path", schema=schema, header=True)

# Window function
from pyspark.sql import Window
import pyspark.sql.functions as F
w = Window.partitionBy("dept").orderBy(F.desc("salary"))
df.withColumn("rank", F.dense_rank().over(w))

# Delta MERGE (Upsert)
from delta.tables import DeltaTable
deltaTable = DeltaTable.forPath(spark, "/delta/events")
deltaTable.alias("tgt").merge(updates.alias("src"), "tgt.id = src.id") \
  .whenMatchedUpdateAll().whenNotMatchedInsertAll().execute()
// YoY Growth %
YoY Growth =
VAR CurrentYear = [Total Sales]
VAR PriorYear = CALCULATE([Total Sales], SAMEPERIODLASTYEAR('Date'[Date]))
RETURN DIVIDE(CurrentYear - PriorYear, PriorYear)

// Running Total
Running Total =
CALCULATE([Total Sales], FILTER(ALL('Date'), 'Date'[Date] <= MAX('Date'[Date])))
